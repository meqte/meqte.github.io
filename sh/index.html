#!/bin/bash ############################################ # Mjack Tools 一键安装脚本 (彻底断开版) ############################################ # 1. 动态获取真实的执行用户和家目录 REAL_USER=${SUDO_USER:-$USER} REAL_HOME=$(getent passwd "$REAL_USER" | cut -d: -f6) # 2. 配置变量 M_URL="https://raw.githubusercontent.com/meqte/meqte.github.io/main/sh/mjack.sh" INSTALL_PATH="${REAL_HOME}/mjack.sh" SHORTCUT_PATH="/usr/local/bin/m" # 颜色定义 GREEN='\033[0;32m' YELLOW='\033[1;33m' RED='\033[0;31m' NC='\033[0m' # 3. 安装流程 echo -e "${GREEN}>>> 正在为用户 ${REAL_USER} 安装 Mjack Tools...${NC}" # 下载逻辑 if curl -sL "$M_URL" -o "$INSTALL_PATH"; then # 确保文件归属于 jack，防止后续运行权限报错 chown "$REAL_USER":"$REAL_USER" "$INSTALL_PATH" 2>/dev/null chmod +x "$INSTALL_PATH" echo -e "${GREEN}✓ 脚本已存至: $INSTALL_PATH${NC}" else echo -e "${RED}✗ 下载失败，请检查网络连接${NC}" exit 1 fi # 创建快捷方式 echo -e "${GREEN}正在创建快捷命令 'm'...${NC}" sudo ln -sf "$INSTALL_PATH" "$SHORTCUT_PATH" hash -r 2>/dev/null # 4. 关键：安装完成即停止，绝不自动运行 echo -e "${GREEN}✓ 安装验证成功！${NC}" echo -e "${YELLOW}==========================================${NC}" echo -e "${YELLOW} 安装已结束。请手动输入以下命令启动： ${NC}" echo -e "${WHITE} m ${NC}" echo -e "${YELLOW}==========================================${NC}" # 强制退出，防止任何后续逻辑执行 exit 0
