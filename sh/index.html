#!/bin/bash ############################################ # Mjack Tools 一键安装脚本 (彻底修复版) ############################################ # 1. 获取执行用户环境 REAL_USER=${SUDO_USER:-$USER} REAL_HOME=$(getent passwd "$REAL_USER" | cut -d: -f6) # 2. 配置变量 M_URL="https://raw.githubusercontent.com/meqte/meqte.github.io/main/sh/mjack.sh" INSTALL_PATH="${REAL_HOME}/mjack.sh" SHORTCUT_PATH="/usr/local/bin/m" # 3. 颜色定义 GREEN='\033[0;32m' YELLOW='\033[1;33m' RED='\033[0;31m' NC='\033[0m' echo -e "${GREEN}>>> 正在为用户 ${REAL_USER} 部署环境...${NC}" # 4. 执行安装 if curl -sL "$M_URL" -o "$INSTALL_PATH"; then # 确保 jack 拥有文件所有权 chown "$REAL_USER":"$REAL_USER" "$INSTALL_PATH" 2>/dev/null chmod +x "$INSTALL_PATH" echo -e "${GREEN}✓ 下载成功: $INSTALL_PATH${NC}" else echo -e "${RED}✗ 下载失败${NC}" exit 1 fi # 创建快捷方式 sudo ln -sf "$INSTALL_PATH" "$SHORTCUT_PATH" # 5. 关键：安装完直接退出，严禁自动启动 echo -e "${GREEN}✓ 安装完成！${NC}" echo -e "${YELLOW}请手动输入 'm' 启动工具箱${NC}" exit 0
