#!/bin/bash ############################################ # Mjack Tools 一键安装脚本 (自适应版本) # bash <(curl -sL meqte.github.io/sh) ############################################ # 1. 动态获取当前真实的执行用户和家目录 # 即使使用 sudo，也能准确找到发起命令的用户家目录 REAL_USER=${SUDO_USER:-$USER} REAL_HOME=$(getent passwd "$REAL_USER" | cut -d: -f6) # 2. 配置变量 M_URL="https://raw.githubusercontent.com/meqte/meqte.github.io/main/sh/mjack.sh" # 将脚本安装在当前用户的家目录下，而不是强制 /root INSTALL_PATH="${REAL_HOME}/mjack.sh" SHORTCUT_PATH="/usr/local/bin/m" VERSION="2026.03.12" # 颜色定义 GREEN='\033[0;32m' BLUE='\033[0;34m' YELLOW='\033[1;33m' RED='\033[0;31m' NC='\033[0m' # 3. 检查权限逻辑修改 # 不再强制要求 root 运行脚本，但在创建快捷方式时会提示需要 sudo if [ "$EUID" -ne 0 ] && [ ! -w "/usr/local/bin" ]; then echo -e "${RED}检测到当前用户对系统路径没有写入权限${NC}" echo -e "${YELLOW}建议使用: sudo bash <(curl -sL meqte.github.io/sh)${NC}" exit 1 fi # 检查并安装 curl if ! command -v curl &> /dev/null; then echo -e "${YELLOW}正在安装 curl...${NC}" # 自动识别 sudo if [ "$EUID" -eq 0 ]; then apt-get update && apt-get install -y curl else sudo apt-get update && sudo apt-get install -y curl fi fi # 4. 下载逻辑 echo -e "${GREEN}正在下载 Mjack Tools 到 $INSTALL_PATH ${NC}" if curl -sL "$M_URL" -o "$INSTALL_PATH"; then # 确保文件所属权属于实际用户而非 root chown "$REAL_USER":"$REAL_USER" "$INSTALL_PATH" 2>/dev/null chmod +x "$INSTALL_PATH" echo -e "${GREEN}✓ 下载成功${NC}" else echo -e "${RED}✗ 下载失败，请检查网络连接${NC}" exit 1 fi # 5. 创建快捷方式 echo -e "${GREEN}正在创建快捷命令 'm'...${NC}" ln -sf "$INSTALL_PATH" "$SHORTCUT_PATH" hash -r 2>/dev/null # 6. 验证安装 if [ -f "$INSTALL_PATH" ]; then echo -e "${GREEN}✓ 安装成功！${NC}" echo -e "${YELLOW}正在自动启动 Mjack Tools...${NC}" # 使用 sudo -u 切换回实际用户运行，防止工具运行在 root 权限下导致生成的配置文件权限错误 sudo -u "$REAL_USER" "$INSTALL_PATH" else echo -e "${RED}✗ 安装验证失败${NC}" exit 1 fi
