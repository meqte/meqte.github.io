#!/bin/bash # ==================================================== # Mjack Tools - RTSP 组件同步安装脚本 # ==================================================== GITHUB_RAW="https://github.com/meqte/meqte.github.io/raw/refs/heads/main/sh/rtsp" TARGET_DIR="$HOME/rtsp" BIN_DIR="$HOME/.local/bin" # 颜色定义 GREEN='\033[0;32m' RED='\033[0;31m' YELLOW='\033[0;33m' NC='\033[0m' echo -e "${GREEN}开始同步 RTSP 组件到 $TARGET_DIR ...${NC}" # 1. 确保目录存在 mkdir -p "$TARGET_DIR/4K" "$TARGET_DIR/logs" "$BIN_DIR" # 2. 组件清单 FILES=("rtsp.sh" "mediamtx" "OK-OK.txt" "4K/1.ts" "mediamtx.yml") # 3. 执行同步 for FILE in "${FILES[@]}"; do DOWNLOAD_URL="${GITHUB_RAW}/${FILE}" echo -e " --> 同步: ${YELLOW}$FILE${NC}" curl -sLf -o "$TARGET_DIR/$FILE" "$DOWNLOAD_URL" if [ $? -ne 0 ]; then echo -e "${RED}[错误] 无法下载: $FILE${NC}" exit 1 fi done # 4. 配置权限 chmod +x "$TARGET_DIR/rtsp.sh" chmod +x "$TARGET_DIR/mediamtx" # 5. 更新软链接 ln -sf "$TARGET_DIR/rtsp.sh" "$BIN_DIR/r" echo -e "${GREEN}==========================================${NC}" echo -e "${GREEN} RTSP 组件同步完成！${NC}" echo -e "${GREEN} 同步路径: $TARGET_DIR${NC}" echo -e "${GREEN} 快捷启动命令: r${NC}" echo -e "${GREEN}==========================================${NC}"
