do DOWNLOAD_URL="${GITHUB_RAW}/${FILE}" echo -e " --> 同步: ${YELLOW}$FILE${NC}" curl -sLf -o "$TARGET_DIR/$FILE" "$DOWNLOAD_URL" if [ $? -ne 0 ]; then echo -e "${RED}[错误] 无法下载: $FILE${NC}" exit 1 fi done # 4. 配置权限 chmod +x "$TARGET_DIR/rtsp.sh" chmod +x "$TARGET_DIR/mediamtx" # ── 自动修复端口绑定权限 (setcap) ──────────────────────────────── echo -e "${YELLOW}正在尝试自动修复端口绑定权限...${NC}" # 检测是否为 OpenWrt 系统 if [ -f /etc/openwrt_release ]; then # OpenWrt 系统：通常不需要 setcap，且不支持 apt-get echo -e "${GREEN}ℹ️ 检测到 OpenWrt 系统，跳过 setcap 权限修复 (通常不需要)${NC}" elif [ "$IS_ROOT" = true ]; then # Root 用户：直接执行，无需 sudo if command -v setcap &> /dev/null; then if setcap cap_net_bind_service=+ep "$TARGET_DIR/mediamtx" 2>/dev/null; then echo -e "${GREEN}✅ 成功赋予 mediamtx 绑定 554 端口权限${NC}" else echo -e "${YELLOW}⚠️ 权限赋予失败，请检查 mediamtx 文件权限。${NC}" fi else # Root 用户但没有 setcap，尝试安装 echo -e "${YELLOW}未找到 setcap 工具，尝试自动安装 libcap2-bin...${NC}" if apt-get update -qq && apt-get install -y -qq libcap2-bin 2>/dev/null; then echo -e "${YELLOW}未找到 setcap 工具，尝试自动安装 libcap2-bin...${NC}" if sudo -n apt-get update -qq && sudo -n apt-get install -y -qq libcap2-bin 2>/dev/null; then echo -e "${GREEN}✅ 已安装 libcap2-bin，正在赋予权限...${NC}" if sudo setcap cap_net_bind_service=+ep "$TARGET_DIR/mediamtx" 2>/dev/null; then echo -e "${GREEN}✅ 成功赋予 mediamtx 绑定 554 端口权限${NC}" else echo -e "${YELLOW}⚠️ 权限赋予失败，请手动执行：sudo setcap cap_net_bind_service=+ep $TARGET_DIR/mediamtx${NC}" fi else echo -e "${YELLOW}⚠️ 自动安装失败（可能需要 sudo 密码）。${NC}" echo -e "${YELLOW} 如需运行 RTSP 服务，请手动执行：${NC}" echo -e "${YELLOW} sudo apt install -y libcap2-bin && sudo setcap cap_net_bind_service=+ep $TARGET_DIR/mediamtx${NC}" fi else echo -e "${YELLOW}⚠️ 未知系统类型，跳过 setcap 修复。${NC}" echo -e "${YELLOW} 如需运行 RTSP 服务，请手动赋予权限或使用高端口。${NC}" fi fi fi # ──────────────────────────────────────────────────────────────────────── # 5. 更新软链接 ln -sf "$TARGET_DIR/rtsp.sh" "$BIN_DIR/r" # ── 自动网络优化 (提升推流质量) ──────────────────────────────────────── echo -e "${YELLOW}正在尝试自动优化网络参数...${NC}" OPT_PARAMS="net.core.rmem_max=33554432\nnet.core.wmem_max=33554432\nnet.core.rmem_default=33554432\nnet.core.wmem_default=33554432\nnet.core.netdev_max_backlog=5000" if [ -f /etc/openwrt_release ]; then # OpenWrt 系统 if [ -f /etc/rc.local ]; then if ! grep -q "net.core.rmem_max" /etc/rc.local 2>/dev/null; then if [ "$IS_ROOT" = true ]; then if sh -c "echo '$OPT_PARAMS' >> /etc/rc.local" 2>/dev/null; then echo -e "${GREEN}✅ 已写入 /etc/rc.local (OpenWrt)${NC}" /etc/init.d/network restart 2>/dev/null || true else echo -e "${YELLOW}⚠️ 自动写入 OpenWrt 配置失败，请手动添加至 /etc/rc.local${NC}" fi else if sudo -n sh -c "echo '$OPT_PARAMS' >> /etc/rc.local" 2>/dev/null; then echo -e "${GREEN}✅ 已写入 /etc/rc.local (OpenWrt)${NC}" sudo /etc/init.d/network restart 2>/dev/null || true else echo -e "${YELLOW}⚠️ 自动写入 OpenWrt 配置失败，请手动添加至 /etc/rc.local${NC}" fi fi else echo -e "${GREEN}ℹ️ 网络参数已优化 (OpenWrt)${NC}" fi else echo -e "${YELLOW}⚠️ 未找到 /etc/rc.local，请手动优化网络参数${NC}" fi elif [ -f /etc/debian_version ]; then # Debian/Ubuntu 系统 CONFIG_FILE="/etc/sysctl.d/99-media-optimize.conf" if [ -f "$CONFIG_FILE" ] && grep -q "net.core.rmem_max" "$CONFIG_FILE" 2>/dev/null; then echo -e "${GREEN}ℹ️ 网络参数已优化 (Debian)${NC}" else if [ "$IS_ROOT" = true ];
